# Getting Started with FastMCP: From Install to Your First Server


## Installation


```bash
# Create a new directory
mkdir -p $HOME/workspace/fastmcp_demo

# Change directory
cd $HOME/workspace/fastmcp_demo
```


```bash
# Start a new virtual environment
uv init --bare --python=3.12

# Install FastMCP
uv add "fastmcp==3.4.2"
```


```bash
uv run fastmcp version
```


## Creating Your First FastMCP Server


```python
from fastmcp import FastMCP

mcp = FastMCP("My MCP Server")
```


```python
@mcp.tool
def sum(a: int, b: int) -> int:
    """Returns the sum of two numbers."""
    return a + b
```


```python
if __name__ == "__main__":
    mcp.run(transport="http", host="localhost", port=8000)
```


```python
if __name__ == "__main__":
    mcp.run()
```


```python
cat > $HOME/workspace/fastmcp_demo/server.py << 'EOF'
from fastmcp import FastMCP

mcp = FastMCP("My MCP Server")

@mcp.tool
def sum(a: int, b: int) -> int:
    """Returns the sum of two numbers."""
    return a + b

if __name__ == "__main__":
    mcp.run(transport="http", host="localhost", port=8000)
EOF
```

## FastMCP JSON Configurations


### Why Use fastmcp.json?


```bash
FASTMCP_HOST=localhost \
  FASTMCP_PORT=8000 \
  FASTMCP_DEBUG=true \
  FASTMCP_SHOW_SERVER_BANNER=false \
  uv run fastmcp run \
  server.py:mcp \
  --transport http \
  --with pandas \
  --with requests
```


### Structure Overview


```json
{
  "$schema": "https://gofastmcp.com/public/schemas/fastmcp.json/v1.json",
  "source": {
    "path": "server.py",
    "entrypoint": "mcp"  // Optional: defaults to auto-detect (mcp/server/app)
  },
  "environment": {  // Optional
    "python": ">=3.12",
    "dependencies": [] // List of packages
  },
  "deployment": {  // Optional
    "transport": "http",
    "host": "localhost",
    "port": 8000,
    "log_level": "INFO"
  }
}
```


#### Source (Required)


```json
"source": {
  "type": "filesystem",  // Optional, default
  "path": "server.py",   // Path to Python file
  "entrypoint": "mcp"    // Variable name (or auto-detect: mcp/server/app)
}
```


#### Environment (Optional)


```json
"environment": {
  "type": "uv",  // Optional, default
  "python": ">=3.10",  // Python version constraint
  "dependencies": ["pandas>=2.0", "requests"],  // Pip packages
  "requirements": "requirements.txt",  // Or use requirements file
  "project": "."  // When using pyproject.toml for dependencies
}
```


#### Deployment (Optional)


```json
"deployment": {
  "transport": "http",  // stdio | http | sse
  "host": "127.0.0.1",  // HTTP only (replaces FASTMCP_HOST)
  "port": 8000,         // HTTP only (replaces FASTMCP_PORT)
  "path": "/mcp",       // Custom endpoint path
  "log_level": "INFO",  // DEBUG | INFO | WARNING | ERROR | CRITICAL (replaces FASTMCP_LOG_LEVEL)
  "env": {              // Environment variables
    "API_KEY": "secret",
    "DATABASE_URL": "postgres://${DB_USER}@${DB_HOST}/db"
  },
  "cwd": "/app",        // Working directory
  "args": ["--config", "server-config.json"]  // Server arguments
}
```

## Testing the Server with a FastMCP Client


```python
import asyncio
from fastmcp import Client

client = Client(<URL>)

async def call_tool(name: str):
    async with client:
        result = await client.call_tool(<tool_name>, <arguments>)
        print(result)

asyncio.run(call_tool(<tool_arguments>))
```


```python
cat > $HOME/workspace/fastmcp_demo/client.py << 'EOF'
import asyncio
from fastmcp import Client

client = Client("http://localhost:8000/mcp")

async def call_sum(a: int, b: int):
    async with client:
        result = await client.call_tool("sum", {"a": a, "b": b})
        print(f"Execution result: {result}")

asyncio.run(call_sum(3, 5))
EOF
```


```bash
# cd $HOME/workspace/fastmcp_demo

# This will run the client
# and call the sum tool with arguments 3 and 5
uv run python client.py
```
