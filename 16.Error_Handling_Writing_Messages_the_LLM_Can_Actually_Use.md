```python
from fastmcp.exceptions import ToolError

@mcp.tool
def dog_to_human_age(
    age: Annotated[int, Field(ge=0, le=30, description="The dog's age in years")],
    breed: Annotated[str, Field(description="The dog's breed")],
) -> int:
    """Calculate the real age of a dog in human years based on its breed."""
    multiplier = get_breed_multiplier(breed)
    if multiplier is None:
        raise ToolError(f"Breed '{breed}' not found. Please provide a valid breed.")
    return age * multiplier
```

## So what does the LLM actually understand?


```python
from fastmcp.exceptions import ToolError

@mcp.tool()
def divide(a: float, b: float) -> float:
    """Divide two numbers."""
    if b == 0:
        # LLM reads: "Cannot divide by zero. Please provide a non-zero divisor."
        raise ToolError("Cannot divide by zero. Please provide a non-zero divisor.")
    return a / b
```


```python
@mcp.tool()
async def fetch_data(url: str) -> str:
    """Fetch data from a URL."""
    try:
        async with httpx.AsyncClient() as client:
            response = await client.get(url)
            response.raise_for_status()
            return response.text
    except httpx.HTTPStatusError as e:
        raise ToolError(f"Request failed with status {e.response.status_code}")
    except httpx.RequestError:
        raise ToolError("Could not connect to the server. Check the URL and try again.")
```


## Masking Internal Error Details


```python
from fastmcp import FastMCP
from fastmcp.exceptions import ToolError

mcp = FastMCP(name="SecureServer", mask_error_details=True)

@mcp.tool()
def risky_operation(param: str) -> str:
    """A tool that might expose sensitive internals."""
    # Imagine this comes from a database driver or file system call
    raise ValueError("Connection failed: host=db.internal.corp, password=s3cret")

@mcp.tool()
def safe_operation(param: str) -> str:
    """A tool with a controlled error message."""
    raise ToolError("Operation failed. Please check your input and try again.")
```


## Resource and Prompt Error Handling


```python
from fastmcp.exceptions import ResourceError

@mcp.resource("config://{name}")
def get_config(name: str) -> str:
    """Read a configuration file by name."""

    config_dir = Path("configs")
    config_file = config_dir / f"{name}.json"

    if not config_file.exists():
        raise ResourceError(f"Configuration '{name}' not found.")

    return config_file.read_text()
```


```python
from fastmcp.exceptions import PromptError

TEMPLATES = {
    "review": "Review the following code for bugs and improvements:\n\n{code}",
    "explain": "Explain this code in simple terms:\n\n{code}",
}

@mcp.prompt()
def code_prompt(template: str, code: str) -> str:
    """Generate a code analysis prompt from a named template."""

    if template not in TEMPLATES:
        raise PromptError(f"Template '{template}' not found. Available: {', '.join(TEMPLATES)}")

    return TEMPLATES[template].format(code=code)
```

from fastmcp.exceptions import ToolError

        # This should succeed — labrador is a known breed

asyncio.run(main())
