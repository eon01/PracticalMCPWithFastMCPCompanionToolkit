## Defining a Lifespan


```python
from fastmcp import FastMCP, Context
from fastmcp.server.lifespan import lifespan

@lifespan
async def app_lifespan(server):
    # This runs once when the server starts
    db = await connect_to_database()
    try:
        yield {"db": db}   # make db available to all tools
    finally:
        # This runs once when the server stops
        await db.close()

mcp = FastMCP("my-server", lifespan=app_lifespan)

@mcp.tool
async def get_user(user_id: int, ctx: Context) -> dict:
    db = ctx.lifespan_context["db"]   # reuse the shared connection
    return await db.fetch_user(user_id)
```
