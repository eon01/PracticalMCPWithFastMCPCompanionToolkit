# FastMCP Middleware: Gates Around Every Request


## The Basic Idea


```text
Request → Middleware A → Middleware B → Your Tool → Middleware B → Middleware A → Response
```


## How to Write One


```python
from fastmcp import FastMCP
from fastmcp.server.middleware import Middleware
from fastmcp.server.middleware import MiddlewareContext

class LoggingMiddleware(Middleware):
    async def on_message(self, context: MiddlewareContext, call_next):
        # This runs before the tool executes
        print(f"Incoming request: {context.method}")

        result = await call_next(context)  # Hand off to the next layer

        # This runs after the tool has returned a result
        print(f"Outgoing response: {context.method}")
        return result

mcp = FastMCP("My Server")

# Add the middleware to the server
mcp.add_middleware(LoggingMiddleware())
```


## Hooks: Choosing the Right Level


## Ordering Matters


```python
mcp.add_middleware(ErrorHandlingMiddleware())  # outermost: catches all errors
mcp.add_middleware(RateLimitingMiddleware())   # runs second on the way in
mcp.add_middleware(LoggingMiddleware())        # innermost: closest to the tool
```


## What Middleware Is Good For


```python
from fastmcp import FastMCP
from fastmcp.server.middleware.logging import LoggingMiddleware

mcp = FastMCP("My Server")
mcp.add_middleware(LoggingMiddleware(include_payloads=True, max_payload_length=1000))
```


```python
from fastmcp.server.middleware.timing import TimingMiddleware

mcp.add_middleware(TimingMiddleware())
```


```python
from fastmcp.server.middleware.rate_limiting import RateLimitingMiddleware

mcp.add_middleware(RateLimitingMiddleware(max_requests_per_second=10.0, burst_capacity=20))
```


```python
from fastmcp.server.middleware.caching import ResponseCachingMiddleware, CallToolSettings

mcp.add_middleware(ResponseCachingMiddleware(
    call_tool_settings=CallToolSettings(ttl=60, included_tools=["expensive_query"])
))
```


```python
from fastmcp.server.middleware.response_limiting import ResponseLimitingMiddleware

# Responses larger than 500 KB are truncated with an explanatory suffix
mcp.add_middleware(ResponseLimitingMiddleware(max_size=500_000))
```


```python
from fastmcp.server.middleware.error_handling import ErrorHandlingMiddleware

mcp.add_middleware(ErrorHandlingMiddleware(include_traceback=True, transform_errors=True))
```


```python
from fastmcp import FastMCP
from fastmcp.server.middleware import Middleware, MiddlewareContext
from fastmcp.exceptions import ToolError

class BlockMiddleware(Middleware):
    async def on_call_tool(self, context: MiddlewareContext, call_next):
        if context.message.name == "my_slow_tool":
            raise ToolError("This tool is disabled for maintenance")
        return await call_next(context)

mcp = FastMCP("My Server")
mcp.add_middleware(BlockMiddleware())
```
